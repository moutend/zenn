extension Array where Element == UInt8 {
    func write(_ path: String) -> Int {
        if !FileManager.default.createFile(atPath: path, contents: nil, attributes: nil) {
            return -1
        }
        guard let fh = FileHandle(forWritingAtPath: path) else {
            return -2
        }

        fh.seekToEndOfFile()
        fh.write(Data(self))
        fh.closeFile()

        return self.count
    }
}
